    <script>
        // Create floating particles
        function createParticles() {
            const particlesContainer = document.getElementById('particles');
            const particleCount = 50;

            for (let i = 0; i < particleCount; i++) {
                const particle = document.createElement('div');
                particle.className = 'particle';
                particle.style.left = Math.random() * 100 + '%';
                particle.style.top = Math.random() * 100 + '%';
                particle.style.animationDelay = Math.random() * 6 + 's';
                particle.style.animationDuration = (Math.random() * 3 + 3) + 's';
                particlesContainer.appendChild(particle);
            }
        }

        // Initialize particles
        createParticles();

        // Get DOM elements
        const usernameInput = document.getElementById('username');
        const getFollowersBtn = document.getElementById('getFollowersBtn');
        const mainForm = document.getElementById('mainForm');
        const loadingContainer = document.getElementById('loadingContainer');
        const successContainer = document.getElementById('successContainer');
        const loadingUsername = document.getElementById('loadingUsername');
        const followerCounter = document.getElementById('followerCounter');
        const getMoreBtn = document.getElementById('getMoreBtn');
        const successUsername = document.getElementById('successUsername');
        const successFollowerCount = document.getElementById('successFollowerCount');
        const statsNumber = document.getElementById('statsNumber');

        let isLoading = false;
        let followerCount = 0;
        let targetFollowers = 0;
        let currentStep = 0;

        // Handle get followers button click
        getFollowersBtn.addEventListener('click', function() {
            const username = usernameInput.value.trim();
            
            if (!username || isLoading) return;
            
            startFollowerProcess(username);
        });

        // Handle get more button click
        getMoreBtn.addEventListener('click', function() {
            resetForm();
        });

        // Handle enter key press
        usernameInput.addEventListener('keypress', function(e) {
            if (e.key === 'Enter') {
                getFollowersBtn.click();
            }
        });

        // Start follower process
        function startFollowerProcess(username) {
            isLoading = true;
            followerCount = 0;
            currentStep = 0;
            targetFollowers = Math.floor(Math.random() * 100) + 150; // Random between 150-250
            
            // Update loading screen
            loadingUsername.textContent = `Processing @${username}`;
            
            // Show loading screen with animation
            loadingContainer.classList.add('active');
            
            // Start follower counter animation
            animateFollowerCounter();
            
            // Animate loading steps
            animateSteps();
            
            // Show success after 10 seconds
            setTimeout(() => {
                showSuccess(username, targetFollowers);
            }, 10000);
        }

        // Animate follower counter
        function animateFollowerCounter() {
            const interval = setInterval(() => {
                if (followerCount < targetFollowers) {
                    const increment = Math.floor(Math.random() * 15) + 5;
                    followerCount = Math.min(followerCount + increment, targetFollowers);
                    followerCounter.textContent = `+${followerCount}`;
                } else {
                    clearInterval(interval);
                }
            }, 200);
        }

        // Animate loading steps
        function animateSteps() {
            const steps = document.querySelectorAll('.step');
            
            setTimeout(() => {
                steps[1].classList.add('active');
            }, 3000);
            
            setTimeout(() => {
                steps[2].classList.add('active');
            }, 8000);
        }

        // Show success screen
        function showSuccess(username, followers) {
            isLoading = false;
            
            // Hide loading screen
            loadingContainer.classList.remove('active');
            
            // Update success content
            successUsername.textContent = `@${username}`;
            successFollowerCount.innerHTML = `<strong>+${followers} followers</strong>`;
            statsNumber.textContent = `+${followers}`;
            
            // Hide main form and show success
            mainForm.style.display = 'none';
            successContainer.classList.add('active');
        }

        // Reset form
        function resetForm() {
            isLoading = false;
            followerCount = 0;
            currentStep = 0;
            usernameInput.value = '';
            
            // Reset loading steps
            const steps = document.querySelectorAll('.step');
            steps.forEach((step, index) => {
                if (index === 0) {
                    step.classList.add('active');
                } else {
                    step.classList.remove('active');
                }
            });
            
            // Hide success and show main form
            successContainer.classList.remove('active');
            mainForm.style.display = 'block';
            
            // Reset counter
            followerCounter.textContent = '+0';
        }

        // Add some interactive effects
        document.addEventListener('mousemove', function(e) {
            const particles = document.querySelectorAll('.particle');
            const mouseX = e.clientX / window.innerWidth;
            const mouseY = e.clientY / window.innerHeight;
            
            particles.forEach((particle, index) => {
                const speed = (index % 5 + 1) * 0.5;
                const x = (mouseX - 0.5) * speed;
                const y = (mouseY - 0.5) * speed;
                particle.style.transform = `translate(${x}px, ${y}px)`;
            });
        });

        // Add input validation and styling
        usernameInput.addEventListener('input', function() {
            const value = this.value;
            const isValid = value.length > 0 && /^[a-zA-Z0-9._]+$/.test(value);
            
            if (isValid || value === '') {
                this.style.borderColor = '#e1e5e9';
            } else {
                this.style.borderColor = '#ff6b6b';
            }
            
            getFollowersBtn.disabled = !isValid;
        });

        // Mobile-specific optimizations
        function isMobileDevice() {
            return (typeof window.orientation !== "undefined") || (navigator.userAgent.indexOf('IEMobile') !== -1);
        }

        if (isMobileDevice()) {
            // Adjust animation speeds for mobile
            document.documentElement.style.setProperty('--animation-speed', '1.2s');
            
            // Prevent zoom on input focus
            usernameInput.addEventListener('focus', function() {
                document.body.style.zoom = "100%";
            });
            
            // Optimize touch events
            document.addEventListener('touchstart', function() {
                // Add touch-specific optimizations
            }, { passive: true });
        }
    </script>
